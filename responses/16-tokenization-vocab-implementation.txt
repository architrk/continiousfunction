```ts
## 1) Tokenization is a segmentation into vocabulary items

A tokenizer maps a string/byte sequence $x$ into tokens $t_1,\\dots,t_n$:

$$x = \\mathrm{concat}(t_1,\\dots,t_n), \\qquad t_i \\in \\mathcal V$$

Then each token becomes an integer ID $\\mathrm{id}(t_i) \\in \\{0,\\dots,|\\mathcal V|-1\\}$ that indexes the embedding table.

## 2) Unigram tokenization as maximum-likelihood segmentation (SentencePiece-style)

Unigram models treat tokenization itself as probabilistic inference: choose the segmentation that maximizes token prior likelihood:

$$\\hat{t}_{1:n} = \\arg\\max_{t:\\;\\mathrm{concat}(t)=x}\\;\\sum_{i=1}^{n} \\log p(t_i)$$

## 3) BPE (byte-pair encoding) builds the vocabulary by merges

BPE starts from base symbols (characters or bytes), then repeatedly merges the most frequent adjacent pair:

$$ (a,b)^* = \\arg\\max_{(a,b)}\\; \\mathrm{count}(a\\,b) \\qquad\\Rightarrow\\qquad a\\;b \\to ab $$

Each merge **increases** vocabulary size (one new token) and usually **decreases** sequence length on texts where that pair is common.

$$\\text{params}_{\\text{token}} \\approx 2\\,|\\mathcal V|\\,d$$

**Tradeoff:** larger vocab can mean **fewer tokens** (cheaper context / KV cache), but larger embedding + output layers (more parameters / memory).
```

```tsx
/**
 * Toy BPE “training” over the input itself:
 * - start from code points (Array.from)
 * - repeatedly merge the most frequent adjacent pair
 * - store each intermediate step for stepping/animation
 */
function computeToyBpeSteps(text: string, maxMerges: number): BpeStep[] {
  const baseTokens = Array.from(text)
  const vocab = new Map<string, number>()
  for (const t of baseTokens) {
    if (!vocab.has(t)) vocab.set(t, vocab.size)
  }

  const steps: BpeStep[] = [{ tokens: baseTokens, vocab: new Map(vocab) }]

  let tokens = baseTokens
  const delim = '\u0000'

  for (let k = 0; k < maxMerges; k++) {
    if (tokens.length < 2) break

    const pairCounts = new Map<string, number>()
    for (let i = 0; i < tokens.length - 1; i++) {
      const key = tokens[i] + delim + tokens[i + 1]
      pairCounts.set(key, (pairCounts.get(key) ?? 0) + 1)
    }

    let bestKey: string | null = null
    let bestCount = 0
    for (const [key, count] of pairCounts.entries()) {
      if (count > bestCount) {
        bestCount = count
        bestKey = key
      }
    }

    // Stop if nothing repeats — no compression left.
    if (!bestKey || bestCount < 2) break

    const splitAt = bestKey.indexOf(delim)
    const a = bestKey.slice(0, splitAt)
    const b = bestKey.slice(splitAt + 1)
    const merged = a + b

    const next: string[] = []
    for (let i = 0; i < tokens.length; ) {
      if (i < tokens.length - 1 && tokens[i] === a && tokens[i + 1] === b) {
        next.push(merged)
        i += 2
      } else {
        next.push(tokens[i])
        i += 1
      }
    }

    tokens = next
    if (!vocab.has(merged)) vocab.set(merged, vocab.size)

    steps.push({
      tokens,
      vocab: new Map(vocab),
      merge: { a, b, merged, count: bestCount }
    })
  }

  return steps
}

function topPairs(tokens: string[], k: number) {
  const delim = '\u0000'
  const pairCounts = new Map<string, number>()
  for (let i = 0; i < tokens.length - 1; i++) {
    const key = tokens[i] + delim + tokens[i + 1]
    pairCounts.set(key, (pairCounts.get(key) ?? 0) + 1)
  }
  const pairs = Array.from(pairCounts.entries())
    .map(([key, count]) => {
      const splitAt = key.indexOf(delim)
      const a = key.slice(0, splitAt)
      const b = key.slice(splitAt + 1)
      return { a, b, merged: a + b, count }
    })
    .sort((x, y) => y.count - x.count)

  return pairs.slice(0, k)
}

/**
 * Toy “unigram” tokenizer:
 * 1) Build a candidate vocab from frequent substrings of the input (plus all single chars).
 * 2) Assign each token a pseudo-probability mass based on frequency * lengthBias.
 * 3) Viterbi DP to find the max-prob segmentation (min -log p).
 */
function computeToyUnigram(
  text: string,
  opts: { maxTokenLength: number; maxVocab: number; lengthBias: number }
): { tokens: string[]; tokenToId: Map<string, number>; tokenToCost: Map<string, number>; vocabSize: number } {
  const chars = Array.from(text)
  const n = chars.length
  if (n === 0) {
    return { tokens: [], tokenToId: new Map(), tokenToCost: new Map(), vocabSize: 0 }
  }

  const { maxTokenLength, maxVocab, lengthBias } = opts

  // Count substrings up to maxTokenLength
  const counts = new Map<string, number>()
  for (let i = 0; i < n; i++) {
    let s = ''
    for (let len = 1; len <= maxTokenLength && i + len <= n; len++) {
      s += chars[i + len - 1]
      counts.set(s, (counts.get(s) ?? 0) + 1)
    }
  }

  // Unique single chars must always exist so DP never gets stuck.
  const singleChars = Array.from(
    new Set(chars)
  )

  // Candidate multi-char tokens: appear at least twice (toy).
  const multi = Array.from(counts.entries())
    .filter(([tok, c]) => tok.length >= 2 && c >= 2)
    .map(([tok, c]) => {
      const len = Array.from(tok).length
      const score = c * Math.pow(len, lengthBias)
      return { tok, c, len, score }
    })
    .sort((a, b) => b.score - a.score)

  const budgetForMulti = Math.max(0, maxVocab - singleChars.length)
  const chosenMulti = multi.slice(0, budgetForMulti)

  // Build final vocab tokens list
  const vocab: Array<{ tok: string; count: number; len: number; mass: number }> = []

  // Add single chars first (so everything is representable).
  for (const ch of singleChars) {
    const c = counts.get(ch) ?? 1
    const len = 1
    const mass = c * Math.pow(len, lengthBias)
    vocab.push({ tok: ch, count: c, len, mass })
  }

  // Add multi tokens
  for (const m of chosenMulti) {
    vocab.push({ tok: m.tok, count: m.c, len: m.len, mass: m.c * Math.pow(m.len, lengthBias) })
  }

  // Convert masses to probabilities
  const totalMass = vocab.reduce((acc, v) => acc + v.mass, 0) || 1

  const tokenToCost = new Map<string, number>()
  for (const v of vocab) {
    const p = v.mass / totalMass
    const cost = -Math.log(Math.max(1e-12, p))
    tokenToCost.set(v.tok, cost)
  }

  // Stable IDs: sort by increasing cost (higher prob) then by length (longer first)
  const sortedByProb = [...vocab]
    .sort((a, b) => {
      const ca = tokenToCost.get(a.tok) ?? 0
      const cb = tokenToCost.get(b.tok) ?? 0
      if (ca !== cb) return ca - cb
      return b.len - a.len
    })
    .map(v => v.tok)

  const tokenToId = new Map<string, number>()
  for (const tok of sortedByProb) {
    if (!tokenToId.has(tok)) tokenToId.set(tok, tokenToId.size)
  }

  // Viterbi DP over code points
  const dp: number[] = new Array(n + 1).fill(Number.POSITIVE_INFINITY)
  const back: Array<{ len: number; tok: string } | null> = new Array(n + 1).fill(null)
  dp[n] = 0

  for (let i = n - 1; i >= 0; i--) {
    let s = ''
    for (let len = 1; len <= maxTokenLength && i + len <= n; len++) {
      s += chars[i + len - 1]
      const costTok = tokenToCost.get(s)
      if (costTok === undefined) continue
      const candidate = costTok + dp[i + len]
      if (candidate < dp[i]) {
        dp[i] = candidate
        back[i] = { len, tok: s }
      }
    }
  }

  // Reconstruct (fallback safety: single-char)
  const out: string[] = []
  for (let i = 0; i < n; ) {
    const choice = back[i]
    if (!choice) {
      out.push(chars[i])
      i += 1
      continue
    }
    out.push(choice.tok)
    i += choice.len
  }

  return {
    tokens: out,
    tokenToId,
    tokenToCost,
    vocabSize: tokenToId.size
  }
}

function computeByteTokens(
  text: string,
  encoder: TextEncoder,
  showInvisibles: boolean
): { tokens: VizToken[]; byteCount: number } {
  const chars = Array.from(text)
  const out: VizToken[] = []

  let byteCount = 0
  for (let i = 0; i < chars.length; i++) {
    const ch = chars[i]
    const bytes = Array.from(encoder.encode(ch))
    byteCount += bytes.length

    for (const b of bytes) {
      let display = ''
      if (b === 0x0a) display = showInvisibles ? '↵' : '\n'
      else if (b === 0x09) display = showInvisibles ? '⇥' : '\t'
      else if (b === 0x0d) display = showInvisibles ? '␍' : '\r'
      else if (b === 0x20) display = showInvisibles ? '·' : ' '
      else if (isPrintableAscii(b)) display = String.fromCharCode(b)
      else display = `\\x${formatHexByte(b)}`

      out.push({
        raw: `0x${formatHexByte(b)}`,
        display,
        id: b,
        meta: `byte ${b} (0x${formatHexByte(b)})`
      })
    }
  }

  return { tokens: out, byteCount }
}

function findSuspicious(text: string): SuspiciousChar[] {
  const hits: SuspiciousChar[] = []
  const byCodePoint = new Map<number, string>()
  for (const item of INVISIBLES) byCodePoint.set(item.codePoint, item.label)

  for (const ch of Array.from(text)) {
    const cp = ch.codePointAt(0) ?? 0
    const label = byCodePoint.get(cp)
    if (label) {
      hits.push({
        ch,
        codePoint: cp,
        hex: `U+${cp.toString(16).toUpperCase().padStart(4, '0')}`,
        label
      })
    }
  }

  // Combining marks are often the “looks same, different bytes/tokens” culprit.
  if (/\p{M}/u.test(text)) {
    hits.push({
      ch: '◌',
      codePoint: -1,
      hex: '—',
      label: 'Combining marks present (Unicode category \\p{M})'
    })
  }

  return hits
}

function formatDollars(x: number) {
  if (!Number.isFinite(x)) return '$0.00'
  if (x < 0.0001) return `$${x.toExponential(2)}`
  return `$${x.toFixed(4)}`
}

function MiniTradeoffChart(props: {
  steps: Array<{ tokenCount: number; vocabSize: number }>
  current: number
}) {
  const { steps, current } = props
  const w = 360
  const h = 120
  const pad = 10

  const n = steps.length
  if (n <= 1) return null

  const tokenCounts = steps.map(s => s.tokenCount)
  const vocabSizes = steps.map(s => s.vocabSize)

  const minTok = Math.min(...tokenCounts)
  const maxTok = Math.max(...tokenCounts)
  const minV = Math.min(...vocabSizes)
  const maxV = Math.max(...vocabSizes)

  const x = (i: number) => pad + (i / (n - 1)) * (w - 2 * pad)
  const yTok = (v: number) => {
    if (maxTok === minTok) return h / 2
    return pad + ((maxTok - v) / (maxTok - minTok)) * (h - 2 * pad)
  }
  const yV = (v: number) => {
    if (maxV === minV) return h / 2
    return pad + ((maxV - v) / (maxV - minV)) * (h - 2 * pad)
  }

  const pathTok = tokenCounts
    .map((v, i) => `${i === 0 ? 'M' : 'L'} ${x(i).toFixed(2)} ${yTok(v).toFixed(2)}`)
    .join(' ')
  const pathV = vocabSizes
    .map((v, i) => `${i === 0 ? 'M' : 'L'} ${x(i).toFixed(2)} ${yV(v).toFixed(2)}`)
    .join(' ')

  const cx = x(clamp(current, 0, n - 1))

  return (
    <div className="rounded-lg border border-slate-800 bg-slate-950/40 p-3">
      <div className="flex items-baseline justify-between gap-3">
        <div className="text-sm font-medium text-slate-200">BPE tradeoff: fewer tokens ↔ bigger vocab</div>
        <div className="text-xs text-slate-400">
          step {current}/{n - 1}
        </div>
      </div>
      <svg viewBox={`0 0 ${w} ${h}`} className="mt-2 h-28 w-full">
        {/* Token count line (down is good) */}
        <path d={pathTok} fill="none" stroke="rgba(96,165,250,0.9)" strokeWidth={2} />
        {/* Vocab size line (up is good) */}
        <path d={pathV} fill="none" stroke="rgba(52,211,153,0.9)" strokeWidth={2} />
        {/* Current step marker */}
        <line x1={cx} x2={cx} y1={pad} y2={h - pad} stroke="rgba(148,163,184,0.35)" strokeWidth={1} />
      </svg>
      <div className="mt-2 flex flex-wrap gap-x-4 gap-y-1 text-xs text-slate-400">
        <div>
          <span className="inline-block h-2 w-2 rounded-sm align-middle" style={{ background: 'rgba(96,165,250,0.9)' }} />{' '}
          tokens (↓)
        </div>
        <div>
          <span className="inline-block h-2 w-2 rounded-sm align-middle" style={{ background: 'rgba(52,211,153,0.9)' }} />{' '}
          vocab size (↑)
        </div>
      </div>
    </div>
  )
}

function TokenStrip(props: {
  tokens: VizToken[]
  showIds: boolean
  showInvisibles: boolean
  renderLimit?: number
}) {
  const { tokens, showIds, showInvisibles, renderLimit = MAX_RENDER_TOKENS } = props

  const clipped = tokens.length > renderLimit
  const shown = clipped ? tokens.slice(0, renderLimit) : tokens

  return (
    <div className="rounded-lg border border-slate-800 bg-slate-950/40 p-3">
      <div className="flex items-baseline justify-between gap-3">
        <div className="text-sm font-medium text-slate-200">Token view</div>
        <div className="text-xs text-slate-400">
          {tokens.length.toLocaleString()} tokens{clipped ? ` (showing first ${renderLimit.toLocaleString()})` : ''}
        </div>
      </div>

      <div className="mt-2 font-mono text-sm leading-relaxed text-slate-100 whitespace-pre-wrap break-words">
        {shown.map((t, i) => {
          const display = makeVisibleForDisplay(t.display, showInvisibles)
          return (
            <span
              key={`${i}-${t.id}-${t.raw}`}
              className="relative inline-block rounded px-1 py-0.5 mr-0.5 mb-0.5 border"
              style={tokenColorStyle(i)}
              title={t.meta ?? t.raw}
            >
              <span>{display}</span>
              {showIds ? (
                <span className="ml-1 text-[10px] text-slate-200/70 align-top">#{t.id}</span>
              ) : null}
            </span>
          )
        })}
      </div>

      {clipped ? (
        <div className="mt-2 text-xs text-amber-300/90">
          Rendering thousands of tiny spans can be slow. Tip: reduce input size, switch mode, or lower BPE merge count.
        </div>
      ) : null}
    </div>
  )
}

  const [rawText, setRawText] = useState<string>(DEFAULT_TEXT)

  const [mode, setMode] = useState<TokenizerMode>('bpe')
  const [normalization, setNormalization] = useState<NormalizationMode>('none')
  const [showInvisibles, setShowInvisibles] = useState<boolean>(true)
  const [showTokenIds, setShowTokenIds] = useState<boolean>(true)

  // Cost lens
  const [pricePer1k, setPricePer1k] = useState<number>(0.01)

  // BPE controls
  const [maxBpeMerges, setMaxBpeMerges] = useState<number>(30)
  const [bpeStep, setBpeStep] = useState<number>(0)
  const [playing, setPlaying] = useState<boolean>(false)
  const playSpeedMs = 450

  // Unigram controls
  const [uniMaxLen, setUniMaxLen] = useState<number>(8)
  const [uniVocab, setUniVocab] = useState<number>(220)
  const [uniLengthBias, setUniLengthBias] = useState<number>(1.15)

  // Debounce input to keep the UI snappy for 1–2k chars
  const [debouncedText, setDebouncedText] = useState<string>(rawText)
  useEffect(() => {
    const t = setTimeout(() => setDebouncedText(rawText), 150)
    return () => clearTimeout(t)
  }, [rawText])

  // Keep a stable TextEncoder instance
  const encoderRef = useRef<TextEncoder | null>(null)
  if (!encoderRef.current) encoderRef.current = new TextEncoder()
  const encoder = encoderRef.current

  const normalizedText = useMemo(() => {
    if (normalization === 'none') return debouncedText
    try {
      return debouncedText.normalize(normalization)
    } catch {
      return debouncedText
    }
  }, [debouncedText, normalization])

  const normalizationPreview = useMemo(() => {
    if (normalization === 'none') return { changed: false, normalized: rawText }
    try {
      const n = rawText.normalize(normalization)
      return { changed: n !== rawText, normalized: n }
    } catch {
      return { changed: false, normalized: rawText }
    }
  }, [rawText, normalization])

  const suspicious = useMemo(() => findSuspicious(normalizedText), [normalizedText])

  const bpeSteps = useMemo(() => {
    if (mode !== 'bpe') return []
    return computeToyBpeSteps(normalizedText, clamp(maxBpeMerges, 0, 80))
  }, [mode, normalizedText, maxBpeMerges])

  // Clamp bpeStep when steps change
  useEffect(() => {
    if (mode !== 'bpe') return
    const maxStep = Math.max(0, bpeSteps.length - 1)
    setBpeStep(s => clamp(s, 0, maxStep))
  }, [mode, bpeSteps.length])

  // Autoplay BPE steps
  useEffect(() => {
    if (mode !== 'bpe') return
    if (!playing) return
    const maxStep = Math.max(0, bpeSteps.length - 1)
    if (maxStep <= 0) return

    const id = window.setInterval(() => {
      setBpeStep(s => {
        if (s >= maxStep) return s
        return s + 1
      })
    }, playSpeedMs)

    return () => window.clearInterval(id)
  }, [mode, playing, bpeSteps.length])

  useEffect(() => {
    if (mode !== 'bpe') return
    const maxStep = Math.max(0, bpeSteps.length - 1)
    if (bpeStep >= maxStep) setPlaying(false)
  }, [mode, bpeStep, bpeSteps.length])

  const bpeViz = useMemo(() => {
    if (mode !== 'bpe') return { tokens: [] as VizToken[], vocabSize: 0, lastMerge: undefined as BpeMerge | undefined }
    const step = bpeSteps[bpeStep] ?? bpeSteps[bpeSteps.length - 1]
    if (!step) return { tokens: [] as VizToken[], vocabSize: 0, lastMerge: undefined as BpeMerge | undefined }
    const tokens: VizToken[] = step.tokens.map((t, i) => ({
      raw: t,
      display: t,
      id: step.vocab.get(t) ?? -1,
      meta: `token "${makeVisibleForDisplay(t, true)}"`
    }))
    return { tokens, vocabSize: step.vocab.size, lastMerge: step.merge }
  }, [mode, bpeSteps, bpeStep])

  const unigramViz = useMemo(() => {
    if (mode !== 'unigram') return { tokens: [] as VizToken[], vocabSize: 0 }
    const res = computeToyUnigram(normalizedText, {
      maxTokenLength: clamp(uniMaxLen, 2, 16),
      maxVocab: clamp(uniVocab, 50, 800),
      lengthBias: clamp(uniLengthBias, 0.5, 2.0)
    })
    const tokens: VizToken[] = res.tokens.map(t => ({
      raw: t,
      display: t,
      id: res.tokenToId.get(t) ?? -1,
      meta: `token "${makeVisibleForDisplay(t, true)}"`
    }))
    return { tokens, vocabSize: res.vocabSize }
  }, [mode, normalizedText, uniMaxLen, uniVocab, uniLengthBias])

  const byteViz = useMemo(() => {
    if (mode !== 'byte') return { tokens: [] as VizToken[], byteCount: 0 }
    return computeByteTokens(normalizedText, encoder, showInvisibles)
  }, [mode, normalizedText, encoder, showInvisibles])

  const activeTokens = useMemo(() => {
    if (mode === 'bpe') return bpeViz.tokens
    if (mode === 'unigram') return unigramViz.tokens
    return byteViz.tokens
  }, [mode, bpeViz.tokens, unigramViz.tokens, byteViz.tokens])

  const charCount = useMemo(() => Array.from(normalizedText).length, [normalizedText])

  const utf8ByteCount = useMemo(() => {
    try {
      return encoder.encode(normalizedText).length
    } catch {
      return normalizedText.length
    }
  }, [encoder, normalizedText])

  const tokenCount = activeTokens.length
  const estimatedCost = (tokenCount / 1000) * (Number.isFinite(pricePer1k) ? pricePer1k : 0)

  const bpeTradeoff = useMemo(() => {
    if (mode !== 'bpe') return null
    const steps = bpeSteps.map(s => ({ tokenCount: s.tokens.length, vocabSize: s.vocab.size }))
    return { steps }
  }, [mode, bpeSteps])

  const currentExample = useMemo(() => EXAMPLES.find(e => e.id === exampleId) ?? EXAMPLES[0], [exampleId])

  // Quick comparison (for the “cost lens”) — compute counts for each mode.
  const compareCounts = useMemo(() => {
    // Keep it cheap: reuse BPE with current max merges, unigram with current settings, and bytes.
    const bpe = computeToyBpeSteps(normalizedText, clamp(maxBpeMerges, 0, 80))
    const bpeLast = bpe[bpe.length - 1]
    const bpeCount = bpeLast?.tokens.length ?? 0

    const uni = computeToyUnigram(normalizedText, {
      maxTokenLength: clamp(uniMaxLen, 2, 16),
      maxVocab: clamp(uniVocab, 50, 800),
      lengthBias: clamp(uniLengthBias, 0.5, 2.0)
    })
    const uniCount = uni.tokens.length

    const bytes = encoder.encode(normalizedText).length

    return { bpeCount, uniCount, byteCount: bytes }
  }, [normalizedText, maxBpeMerges, uniMaxLen, uniVocab, uniLengthBias, encoder])

  return (
    <div className="w-full">
      <div className="rounded-xl border border-slate-800 bg-slate-950/40 p-4">
        <div className="flex flex-col gap-1">
          <div className="text-lg font-semibold text-slate-100">Tokenization microscope</div>
          <div className="text-sm text-slate-400">
            A self-contained, <span className="text-slate-300">toy</span> tokenizer lab: watch BPE merges build a vocabulary, see unigram
            segmentation choose the most likely path, and see byte-level explode non-ASCII into bytes.
          </div>
        </div>

        <div className="mt-4 grid grid-cols-1 gap-4 lg:grid-cols-2">
          {/* Left: input + controls */}
          <div className="space-y-4">
            <div className="rounded-lg border border-slate-800 bg-slate-950/40 p-3">
              <div className="flex flex-wrap items-center justify-between gap-2">
                <div className="text-sm font-medium text-slate-200">Curated examples</div>
                <div className="flex flex-wrap gap-2">
                  {EXAMPLES.map(ex => (
                    <button
                      key={ex.id}
                      onClick={() => {
                        setExampleId(ex.id)
                        setRawText(ex.text)
                        setPlaying(false)
                        setBpeStep(0)
                      }}
                      className={[
                        'rounded-md px-2 py-1 text-xs border transition',
                        ex.id === exampleId
                          ? 'border-slate-400 bg-slate-800/60 text-slate-100'
                          : 'border-slate-800 bg-slate-900/30 text-slate-300 hover:bg-slate-800/40'
                      ].join(' ')}
                    >
                      {ex.label}
                    </button>
                  ))}
                </div>
              </div>
              <div className="mt-2 text-xs text-slate-400">{currentExample?.description}</div>
            </div>

            <div className="rounded-lg border border-slate-800 bg-slate-950/40 p-3">
              <div className="flex items-center justify-between gap-3">
                <div className="text-sm font-medium text-slate-200">Input text</div>
                <div className="text-xs text-slate-400">
                  {Array.from(rawText).length.toLocaleString()} chars
                </div>
              </div>

              <textarea
                value={rawText}
                onChange={e => {
                  setRawText(e.target.value)
                  setPlaying(false)
                }}
                spellCheck={false}
                className="mt-2 h-40 w-full resize-y rounded-md border border-slate-800 bg-slate-900/40 p-2 font-mono text-sm text-slate-100 outline-none focus:border-slate-500"
              />

              <div className="mt-3 grid grid-cols-1 gap-3 md:grid-cols-2">
                <div>
                  <label className="block text-xs text-slate-400">Normalization</label>
                  <select
                    value={normalization}
                    onChange={e => setNormalization(e.target.value as NormalizationMode)}
                    className="mt-1 w-full rounded-md border border-slate-800 bg-slate-900/40 px-2 py-1 text-sm text-slate-100 outline-none focus:border-slate-500"
                  >
                    <option value="none">None</option>
                    <option value="NFC">NFC</option>
                    <option value="NFKC">NFKC</option>
                  </select>
                  {normalization !== 'none' ? (
                    <div className="mt-1 text-xs text-slate-400">
                      {normalizationPreview.changed ? (
                        <span className="text-amber-300/90">
                          Changed by normalization (try the “Invisible Unicode” example).
                        </span>
                      ) : (
                        <span>Text unchanged under {normalization}.</span>
                      )}
                    </div>
                  ) : null}
                </div>

                <div>
                  <label className="block text-xs text-slate-400">Cost per 1k tokens ($)</label>
                  <input
                    type="number"
                    value={Number.isFinite(pricePer1k) ? pricePer1k : 0}
                    min={0}
                    step={0.001}
                    onChange={e => setPricePer1k(Number(e.target.value))}
                    className="mt-1 w-full rounded-md border border-slate-800 bg-slate-900/40 px-2 py-1 text-sm text-slate-100 outline-none focus:border-slate-500"
                  />
                  <div className="mt-1 text-xs text-slate-400">Used only for the “cost lens” estimate.</div>
                </div>
              </div>

              <div className="mt-3 flex flex-wrap items-center gap-3">
                <label className="flex items-center gap-2 text-xs text-slate-300">
                  <input
                    type="checkbox"
                    checked={showInvisibles}
                    onChange={e => setShowInvisibles(e.target.checked)}
                    className="h-4 w-4 rounded border-slate-700 bg-slate-900"
                  />
                  Show invisible/whitespace markers
                </label>
                <label className="flex items-center gap-2 text-xs text-slate-300">
                  <input
                    type="checkbox"
                    checked={showTokenIds}
                    onChange={e => setShowTokenIds(e.target.checked)}
                    className="h-4 w-4 rounded border-slate-700 bg-slate-900"
                  />
                  Show token IDs
                </label>
              </div>
            </div>

            <div className="rounded-lg border border-slate-800 bg-slate-950/40 p-3">
              <div className="text-sm font-medium text-slate-200">Tokenizer mode</div>
              <div className="mt-2 flex flex-wrap gap-2">
                {([
                  { id: 'bpe', label: 'Toy BPE merges' },
                  { id: 'unigram', label: 'Toy unigram segmentation' },
                  { id: 'byte', label: 'Byte-level (UTF-8)' }
                ] as Array<{ id: TokenizerMode; label: string }>).map(opt => (
                  <button
                    key={opt.id}
                    onClick={() => {
                      setMode(opt.id)
                      setPlaying(false)
                      setBpeStep(0)
                    }}
                    className={[
                      'rounded-md px-3 py-1.5 text-sm border transition',
                      mode === opt.id
                        ? 'border-slate-400 bg-slate-800/60 text-slate-100'
                        : 'border-slate-800 bg-slate-900/30 text-slate-300 hover:bg-slate-800/40'
                    ].join(' ')}
                  >
                    {opt.label}
                  </button>
                ))}
              </div>

              {mode === 'bpe' ? (
                <div className="mt-4 space-y-3">
                  <div className="grid grid-cols-1 gap-3 md:grid-cols-2">
                    <div>
                      <label className="block text-xs text-slate-400">Max merges to compute</label>
                      <input
                        type="range"
                        min={0}
                        max={80}
                        value={maxBpeMerges}
                        onChange={e => {
                          setMaxBpeMerges(Number(e.target.value))
                          setPlaying(false)
                          setBpeStep(0)
                        }}
                        className="mt-2 w-full"
                      />
                      <div className="mt-1 text-xs text-slate-400">{maxBpeMerges} merges</div>
                    </div>
                    <div>
                      <label className="block text-xs text-slate-400">Step through merges</label>
                      <input
                        type="range"
                        min={0}
                        max={Math.max(0, bpeSteps.length - 1)}
                        value={bpeStep}
                        onChange={e => {
                          setBpeStep(Number(e.target.value))
                          setPlaying(false)
                        }}
                        className="mt-2 w-full"
                        disabled={bpeSteps.length <= 1}
                      />
                      <div className="mt-1 text-xs text-slate-400">
                        step {bpeStep}/{Math.max(0, bpeSteps.length - 1)}
                      </div>
                    </div>
                  </div>

                  <div className="flex flex-wrap items-center gap-2">
                    <button
                      onClick={() => {
                        setPlaying(false)
                        setBpeStep(0)
                      }}
                      className="rounded-md border border-slate-800 bg-slate-900/30 px-2 py-1 text-xs text-slate-200 hover:bg-slate-800/40"
                    >
                      Reset
                    </button>
                    <button
                      onClick={() => {
                        setPlaying(false)
                        setBpeStep(s => clamp(s - 1, 0, Math.max(0, bpeSteps.length - 1)))
                      }}
                      className="rounded-md border border-slate-800 bg-slate-900/30 px-2 py-1 text-xs text-slate-200 hover:bg-slate-800/40"
                      disabled={bpeStep <= 0}
                    >
                      ◀ Step
                    </button>
                    <button
                      onClick={() => {
                        setPlaying(false)
                        setBpeStep(s => clamp(s + 1, 0, Math.max(0, bpeSteps.length - 1)))
                      }}
                      className="rounded-md border border-slate-800 bg-slate-900/30 px-2 py-1 text-xs text-slate-200 hover:bg-slate-800/40"
                      disabled={bpeStep >= Math.max(0, bpeSteps.length - 1)}
                    >
                      Step ▶
                    </button>
                    <button
                      onClick={() => setPlaying(p => !p)}
                      className={[
                        'rounded-md border px-2 py-1 text-xs',
                        playing
                          ? 'border-emerald-400/50 bg-emerald-500/10 text-emerald-200'
                          : 'border-slate-800 bg-slate-900/30 text-slate-200 hover:bg-slate-800/40'
                      ].join(' ')}
                      disabled={bpeSteps.length <= 1}
                    >
                      {playing ? 'Pause' : 'Play'}
                    </button>
                  </div>

                  {bpeTradeoff ? <MiniTradeoffChart steps={bpeTradeoff.steps} current={bpeStep} /> : null}

                  <div className="rounded-lg border border-slate-800 bg-slate-900/20 p-3">
                    <div className="text-xs text-slate-300">
                      {bpeViz.lastMerge ? (
                        <>
                          <span className="text-slate-400">Last merge:</span>{' '}
                          <span className="font-mono">
                            {makeVisibleForDisplay(bpeViz.lastMerge.a, true)} + {makeVisibleForDisplay(bpeViz.lastMerge.b, true)} →{' '}
                            {makeVisibleForDisplay(bpeViz.lastMerge.merged, true)}
                          </span>{' '}
                          <span className="text-slate-400">(count {bpeViz.lastMerge.count})</span>
                        </>
                      ) : (
                        <span className="text-slate-400">Step 0: start from characters (code points).</span>
                      )}
                    </div>

                    <div className="mt-2">
                      <div className="text-xs font-medium text-slate-200">Top adjacent pairs right now</div>
                      <div className="mt-1 grid grid-cols-1 gap-1 sm:grid-cols-2">
                        {topPairs((bpeSteps[bpeStep]?.tokens ?? []), 6).map((p, i) => (
                          <div key={i} className="text-xs text-slate-300">
                            <span className="font-mono">
                              {makeVisibleForDisplay(p.a, true)}·{makeVisibleForDisplay(p.b, true)}
                            </span>{' '}
                            <span className="text-slate-400">× {p.count}</span>
                          </div>
                        ))}
                      </div>
                    </div>
                  </div>

                  <div className="text-xs text-slate-400">
                    Aha: each merge grows the vocabulary (one new token) and typically reduces sequence length for frequent patterns.
                  </div>
                </div>
              ) : null}

              {mode === 'unigram' ? (
                <div className="mt-4 grid grid-cols-1 gap-3 md:grid-cols-3">
                  <div>
                    <label className="block text-xs text-slate-400">Max token length</label>
                    <input
                      type="range"
                      min={2}
                      max={16}
                      value={uniMaxLen}
                      onChange={e => setUniMaxLen(Number(e.target.value))}
                      className="mt-2 w-full"
                    />
                    <div className="mt-1 text-xs text-slate-400">{uniMaxLen}</div>
                  </div>
                  <div>
                    <label className="block text-xs text-slate-400">Vocab size cap</label>
                    <input
                      type="range"
                      min={50}
                      max={800}
                      step={10}
                      value={uniVocab}
                      onChange={e => setUniVocab(Number(e.target.value))}
                      className="mt-2 w-full"
                    />
                    <div className="mt-1 text-xs text-slate-400">{uniVocab}</div>
                  </div>
                  <div>
                    <label className="block text-xs text-slate-400">Length bias</label>
                    <input
                      type="range"
                      min={0.5}
                      max={2.0}
                      step={0.05}
                      value={uniLengthBias}
                      onChange={e => setUniLengthBias(Number(e.target.value))}
                      className="mt-2 w-full"
                    />
                    <div className="mt-1 text-xs text-slate-400">{uniLengthBias.toFixed(2)}</div>
                  </div>

                  <div className="md:col-span-3 text-xs text-slate-400">
                    Aha: unigram tokenization is inference — it chooses the segmentation that maximizes token prior likelihood, not “words.”
                  </div>
                </div>
              ) : null}

              {mode === 'byte' ? (
                <div className="mt-4 text-xs text-slate-400">
                  Byte-level mode uses UTF-8 bytes as tokens (IDs 0–255). Non-ASCII characters expand into multiple bytes.
                </div>
              ) : null}
            </div>

            {/* Warnings / gotchas */}
            <div className="rounded-lg border border-slate-800 bg-slate-950/40 p-3">
              <div className="text-sm font-medium text-slate-200">Unicode / safety gotchas</div>
              {suspicious.length === 0 ? (
                <div className="mt-2 text-xs text-slate-400">No common invisible characters detected in the (normalized) text.</div>
              ) : (
                <div className="mt-2 space-y-1">
                  {suspicious.map((s, idx) => (
                    <div key={idx} className="text-xs text-slate-300">
                      <span className="font-mono text-slate-100">{s.hex}</span> — {s.label}
                    </div>
                  ))}
                </div>
              )}
              <div className="mt-3 text-xs text-slate-400">
                Tip: try toggling NFC/NFKC and “Show invisible markers” on the <span className="text-slate-300">Invisible Unicode</span> example.
              </div>
            </div>
          </div>

          {/* Right: results */}
          <div className="space-y-4">
            {/* Stats */}
            <div className="rounded-lg border border-slate-800 bg-slate-950/40 p-3">
              <div className="text-sm font-medium text-slate-200">Token & cost lens</div>

              <div className="mt-2 grid grid-cols-2 gap-3 text-sm">
                <div className="rounded-md border border-slate-800 bg-slate-900/20 p-2">
                  <div className="text-xs text-slate-400">Mode</div>
                  <div className="mt-1 text-slate-100">
                    {mode === 'bpe' ? 'Toy BPE' : mode === 'unigram' ? 'Toy unigram' : 'Byte-level'}
                  </div>
                </div>

                <div className="rounded-md border border-slate-800 bg-slate-900/20 p-2">
                  <div className="text-xs text-slate-400">Tokens</div>
                  <div className="mt-1 text-slate-100">{tokenCount.toLocaleString()}</div>
                </div>

                <div className="rounded-md border border-slate-800 bg-slate-900/20 p-2">
                  <div className="text-xs text-slate-400">Characters</div>
                  <div className="mt-1 text-slate-100">{charCount.toLocaleString()}</div>
                </div>

                <div className="rounded-md border border-slate-800 bg-slate-900/20 p-2">
                  <div className="text-xs text-slate-400">UTF-8 bytes</div>
                  <div className="mt-1 text-slate-100">{utf8ByteCount.toLocaleString()}</div>
                </div>

                <div className="rounded-md border border-slate-800 bg-slate-900/20 p-2">
                  <div className="text-xs text-slate-400">Estimated cost</div>
                  <div className="mt-1 text-slate-100">{formatDollars(estimatedCost)}</div>
                  <div className="mt-0.5 text-[11px] text-slate-400">${pricePer1k}/1k tokens</div>
                </div>

                <div className="rounded-md border border-slate-800 bg-slate-900/20 p-2">
                  <div className="text-xs text-slate-400">Vocab size (toy)</div>
                  <div className="mt-1 text-slate-100">
                    {mode === 'bpe' ? bpeViz.vocabSize.toLocaleString() : mode === 'unigram' ? unigramViz.vocabSize.toLocaleString() : '256'}
                  </div>
                </div>
              </div>

              <div className="mt-3 rounded-md border border-slate-800 bg-slate-900/20 p-2">
                <div className="text-xs font-medium text-slate-200">Quick compare (same text)</div>
                <div className="mt-2 grid grid-cols-3 gap-2 text-xs text-slate-300">
                  <div className="rounded border border-slate-800 bg-slate-950/30 p-2">
                    <div className="text-slate-400">Toy BPE</div>
                    <div className="mt-1 text-slate-100">{compareCounts.bpeCount.toLocaleString()} tokens</div>
                  </div>
                  <div className="rounded border border-slate-800 bg-slate-950/30 p-2">
                    <div className="text-slate-400">Toy unigram</div>
                    <div className="mt-1 text-slate-100">{compareCounts.uniCount.toLocaleString()} tokens</div>
                  </div>
                  <div className="rounded border border-slate-800 bg-slate-950/30 p-2">
                    <div className="text-slate-400">Byte-level</div>
                    <div className="mt-1 text-slate-100">{compareCounts.byteCount.toLocaleString()} bytes</div>
                  </div>
                </div>
                <div className="mt-2 text-xs text-slate-400">
                  Aha: “context length” and cost are about tokens. Different tokenizers can make the *same visible text* much cheaper or more expensive.
                </div>
              </div>

              {mode === 'byte' ? (
                <div className="mt-2 text-xs text-slate-400">
                  Byte-level shows <span className="text-slate-300">bytes</span> as tokens (not characters). Multi-byte Unicode often expands.
                </div>
              ) : null}

              {mode === 'bpe' ? (
                <div className="mt-2 text-xs text-slate-400">
                  This toy BPE “learns” merges from your input (real tokenizers learn from huge corpora, so merges reflect training distribution).
                </div>
              ) : null}

              {mode === 'unigram' ? (
                <div className="mt-2 text-xs text-slate-400">
                  This toy unigram builds a small vocab from frequent substrings and uses Viterbi segmentation (real unigram models learn token priors).
                </div>
              ) : null}
            </div>

            {/* Token view */}
            <TokenStrip tokens={activeTokens} showIds={showTokenIds} showInvisibles={showInvisibles} />

            {/* Byte-only detail */}
            {mode === 'byte' ? (
              <div className="rounded-lg border border-slate-800 bg-slate-950/40 p-3">
                <div className="text-sm font-medium text-slate-200">Byte-level note</div>
                <div className="mt-2 text-xs text-slate-400">
                  In byte-level mode, each token ID is literally the byte value 0–255. Notice how emoji and many non-Latin scripts use multiple bytes.
                </div>
                <div className="mt-2 text-xs text-slate-400">
                  Total bytes for this text: <span className="text-slate-100">{byteViz.byteCount.toLocaleString()}</span>
                </div>
              </div>
            ) : null}

            {/* Normalization note */}
            {normalization !== 'none' && normalizationPreview.changed ? (
              <div className="rounded-lg border border-amber-500/30 bg-amber-500/10 p-3">
                <div className="text-sm font-medium text-amber-200">Normalization changed the text</div>
                <div className="mt-2 text-xs text-amber-100/90">
                  Visually similar strings can have different code points (and bytes) → different tokens. This matters for reliability, cost, and security.
                </div>
              </div>
            ) : null}
          </div>
        </div>
      </div>
    </div>
  )
}
```
